## vim:set ft=mako:
## intended to be used from an all species search. (i.e. "species like %")
<%
all_strings = set()

strings = set()
for species in values:
  if species.sp:
    for i in species.sp.split():
      strings.add(i)
  if species.infrasp1:
    for i in species.infrasp1.split():
      strings.add(i)
  if species.infrasp2:
    for i in species.infrasp1.split():
      strings.add(i)
  if species.infrasp3:
    for i in species.infrasp1.split():
      strings.add(i)
  if species.infrasp4:
    for i in species.infrasp1.split():
      strings.add(i)
  strings.add(species.genus.genus)
  strings.add(species.genus.family.family)
for i in strings:
  i = i.strip(' ()&.')
  if len(i) > 1:
    all_strings.add(i)
%>\

${part}
% endfor
